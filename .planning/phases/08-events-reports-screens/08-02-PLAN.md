```swift
struct Report: Codable, Identifiable, Sendable {
    let id: String
    let userId: String
    let websiteId: String
    let type: String         // "funnel", "utm", "goals", "attribution"
    let name: String
    let description: String?
    let parameters: String?  // JSON string
    let createdAt: String
    let updatedAt: String?
}

struct FunnelStep: Codable, Identifiable, Sendable {
    let type: String       // "path" or "event"
    let value: String
    let visitors: Int
    let dropoff: Int
    var dropoffRate: Double { ... }
}

struct UTMReportItem: Codable, Identifiable, Sendable {
    let source: String?
    let medium: String?
    let campaign: String?
    let content: String?
    let term: String?
    let visitors: Int
}

struct GoalReportItem: Codable, Identifiable, Sendable {
    let type: String       // "path" or "event"
    let value: String
    let goal: Int
    let result: Int
    var completionRate: Double { ... }
}

struct AttributionItem: Codable, Identifiable, Sendable {
    let source: String?
    let medium: String?
    let campaign: String?
    let channel: String?
    let visitors: Int
}
```

From UmamiAPI (relevant methods):
```swift
func getReports(websiteId:, page:, pageSize:) -> ReportListResponse
func getFunnelReport(websiteId:, dateRange:, steps: [[String: String]]) -> [FunnelStep]
func getUTMReport(websiteId:, dateRange:) -> [UTMReportItem]
func getGoalReport(websiteId:, dateRange:, goals: [[String: Any]]) -> [GoalReportItem]
func getAttributionReport(websiteId:, dateRange:) -> [AttributionItem]
```

Note: getFunnelReport braucht `steps` Parameter und getGoalReport braucht `goals` Parameter.
Diese muessen aus den Report-parameters (JSON string) geparsed werden.
Alternativ: getReports() laden und die gespeicherten Reports mit ihren Parametern verwenden.

```
struct ReportsHubView: View {
    let website: Website
    @StateObject private var viewModel: ReportsViewModel
    @State private var selectedDateRange: DateRange = .thisMonth
```

1. DateRange-Picker oben (wie SessionsView)
2. LazyVGrid (2 Spalten) mit 4 Report-Karten:
   - Funnel: icon "chart.bar.doc.horizontal", Farbe .blue, Titel "Funnels", Untertitel "Conversion-Pfade"
   - UTM: icon "link.badge.plus", Farbe .green, Titel "UTM-Kampagnen", Untertitel "Traffic-Quellen"
   - Goals: icon "target", Farbe .orange, Titel "Goals", Untertitel "Ziel-Tracking"
   - Attribution: icon "point.3.filled.connected.trianglepath.dotted", Farbe .purple, Titel "Attribution", Untertitel "Kanal-Zuordnung"

Jede Karte: GlassCard-Style (VStack: Icon oben links, Titel, Untertitel, chevron.right), NavigationLink zum jeweiligen Detail-View.

Task: `.task { await viewModel.loadReports() }` — Laedt Report-Liste fuer Funnel/Goal-Parameter

**ReportDetailViews.swift** — 4 separate Views:

1. **UTMReportView(website: Website, dateRange: DateRange)**:
   - @StateObject ReportsViewModel
   - Task: loadUTMReport(dateRange:)
   - Layout: Liste mit UTMReportItem-Zeilen. Jede Zeile: Source/Medium/Campaign als Tags, Visitors als Badge rechts
   - Gruppierung: Nach source sortiert
   - Empty State: ContentUnavailableView "Keine UTM-Daten"

2. **AttributionReportView(website: Website, dateRange: DateRange)**:
   - @StateObject ReportsViewModel
   - Task: loadAttributionReport(dateRange:)
   - Layout: Liste mit AttributionItem-Zeilen. Jede Zeile: Channel/Source, Visitors-Badge
   - Empty State: ContentUnavailableView "Keine Attribution-Daten"

3. **FunnelReportView(website: Website, reports: [Report], dateRange: DateRange)**:
   - Braucht gespeicherte Funnel-Reports fuer steps-Parameter
   - Falls keine Funnel-Reports existieren: Info-Text "Keine Funnel-Reports konfiguriert. Erstelle Funnel-Reports im Umami Dashboard."
   - Falls Funnel-Reports vorhanden: Picker fuer Report-Auswahl, dann Steps als vertikale Pipeline:
     - Jeder Step: Breiter Balken (Breite proportional zu visitors/maxVisitors), Step-Name, Visitors-Count, Dropoff-Rate als roter Text
   - Task: loadFunnelReport(report:dateRange:)

4. **GoalReportView(website: Website, reports: [Report], dateRange: DateRange)**:
   - Braucht gespeicherte Goal-Reports fuer goals-Parameter
   - Falls keine Goal-Reports: Info-Text "Keine Goal-Reports konfiguriert."
   - Falls vorhanden: Picker + Goals als Liste mit ProgressView-Balken:
     - Goal-Name, Result/Goal als Fraction, completionRate als Prozent, ProgressView als visueller Balken
   - Task: loadGoalReport(report:dateRange:)
