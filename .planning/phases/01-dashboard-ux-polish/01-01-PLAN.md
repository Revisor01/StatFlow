**WICHTIG fuer Checkmark-Darstellung:** SwiftUI Menu unterstuetzt keine direkte Checkmark-Darstellung auf Buttons. Stattdessen Picker mit `.pickerStyle(.menu)` verwenden oder den aktiven Account-Namen mit einem Checkmark-Emoji/Symbol im Label kennzeichnen. Bevorzugte Loesung: **Picker in Menu verwenden**, da SwiftUI Picker innerhalb eines Menu automatisch den ausgewaehlten Wert mit Checkmark markiert:

```swift
Menu {
    // Picker fuer automatische Checkmark-Markierung
    Picker(selection: Binding(
        get: { accountManager.activeAccount?.id ?? "" },
        set: { newId in
            if let account = accountManager.accounts.first(where: { $0.id == newId }) {
                Task {
                    await accountManager.setActiveAccount(account)
                    await viewModel.loadData(dateRange: selectedDateRange)
                }
            }
        }
    ), label: Text("Account")) {
        ForEach(accountManager.accounts) { account in
            Label {
                Text(account.displayName)
            } icon: {
                Image(systemName: account.icon)
            }
            .tag(account.id)
        }
    }

    Divider()

    Button {
        showingAddAccount = true
    } label: {
        Label("account.switcher.addAccount", systemImage: "plus.circle.fill")
    }
} label: {
    Image(systemName: accountManager.activeAccount?.icon ?? "server.rack")
        .foregroundStyle(
            accountManager.activeAccount?.providerType == .umami ? .orange : .blue
        )
}
```
