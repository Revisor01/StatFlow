LoginViewModel uebernimmt die Login-Logik 1:1 aus AuthManager:

```swift
import Foundation
import Combine

@MainActor
class LoginViewModel: ObservableObject {
    @Published var isLoading = false
    @Published var errorMessage: String?

    /// Umami Login — Code aus AuthManager.login() uebernommen
    func login(serverURL: String, username: String, password: String, accountName: String = "") async {
        isLoading = true
        errorMessage = nil
        defer { isLoading = false }

        do {
            // 1. Authenticate via UmamiAPI
            try await UmamiAPI.shared.authenticate(
                serverURL: serverURL,
                credentials: .usernamePassword(username: username, password: password)
            )

            // 2. Account erstellen und in AccountManager speichern
            let account = AnalyticsAccount(
                id: UUID().uuidString,
                name: accountName.isEmpty ? username : accountName,
                serverURL: serverURL,
                providerType: .umami,
                credentials: .usernamePassword(username: username, password: ""), // Passwort geht in Keychain
                sites: nil
            )
            AccountManager.shared.addAccount(account)
            AccountManager.shared.setActiveAccount(account)
        } catch {
            errorMessage = error.localizedDescription
        }
    }

    /// Plausible Login — Code aus AuthManager.loginWithPlausible() uebernommen
    func loginWithPlausible(serverURL: String, apiKey: String, accountName: String = "") async {
        isLoading = true
        errorMessage = nil
        defer { isLoading = false }

        do {
            // 1. Authenticate via PlausibleAPI
            try await PlausibleAPI.shared.authenticate(
                serverURL: serverURL,
                credentials: .apiKey(apiKey)
            )

            // 2. Account erstellen
            let account = AnalyticsAccount(
                id: UUID().uuidString,
                name: accountName.isEmpty ? "Plausible" : accountName,
                serverURL: serverURL,
                providerType: .plausible,
                credentials: .apiKey(""), // API-Key geht in Keychain
                sites: nil
            )
            AccountManager.shared.addAccount(account)
            AccountManager.shared.setActiveAccount(account)
        } catch {
            errorMessage = error.localizedDescription
        }
    }
}
```
